49/**
50 * @brief Read a BEJ non-negative integer (nnint) as per DSP0218.
51 *
52 * Encoding is: a single length byte N, followed by N bytes containing
53 * a little-endian unsigned integer value.
54 *
55 * @param b Reader positioned at the start of an nnint.
56 * @param out Output decoded value.
57 * @return 1 on success, 0 on malformed/overflow.
58 */
59int bej_read_nnint(bej_br* b, uint64_t* out){
60 uint8_t N; if(!bej_br_u8(b,&N)) return 0;
61 uint64_t v=0;
62 if(b->p+N > b->n) return 0;
63 for(unsigned i=0;i<N;i++) v |= (uint64_t)b->d[b->p+i] << (8*i);
64 b->p += N;
65 *out = v;
66 return 1;
67}
