1/**
2 * @file bej_decode.c
3 * @brief Minimal BEJ decoder (subset) per DMTF DSP0218 suitable for the provided example.
4 *
5 * This implementation:
6 * - Parses a Redfish **schema dictionary** binary (Table 31) and exposes a map of
7 * sequence numbers to property names and child clusters.
8 * - Decodes a BEJ **bejEncoding** stream (version, flags, schema class) followed by a top-level tuple.
9 * - Supports value formats: **Set**, **Array**, **Integer**, **String**.
10 * - **Annotations** are **ignored/skipped** by design (per task requirement).
11 * - **Enum** values are rendered as strings (resolved via the dictionary options cluster).
12 * - Emits pretty-printed JSON to the output file.
13 *
14 * @note This is a pragmatic subset intended to match the task's example.
15 * It does **not** implement every BEJ/Redfish type or all validation rules in DSP0218.
16 */

24static int decode_value_int(bej_jsonw* jw, bej_br* br, uint64_t L){
25 long long v=0;
26 for(uint64_t i=0;i<L;i++){
27 uint8_t b; if(!bej_br_u8(br,&b)) return 0;
28 v |= (long long)b << (8*i);
29 }
30 bej_jw_int(jw, v);
31 return 1;
32}

34static int decode_value_string(bej_jsonw* jw, bej_br* br, uint64_t L){
35 char* s=(char*)malloc((size_t)L+1); if(!s) return 0;
36 if(!bej_br_get(br, (uint8_t*)s, (size_t)L)){ free(s); return 0; }
37 s[L]=0;
38 /* strip trailing NUL(s) for neat JSON */
39 for(long i=(long)L-1; i>=0; --i){ if(s[i]==0) s[i]=0; else break; }
40 bej_jw_str(jw, s);
41 free(s);
42 return 1;
43}

48/**
49 * @brief Decode a Set value: a count followed by that many tuples; emit a JSON object.
50 *
51 * @param jw JSON writer.
52 * @param br Reader positioned at the Set value (expects a leading nnint count).
53 * @param D Parsed dictionary (for names and child clusters).
54 * @param this_cluster The cluster that defines member sequence numbers for this Set.
55 * @return 1 on success, 0 on malformed input.
56 *
57 * @note Annotations (S LSB bit set) are skipped entirely per task requirement.
58 */
59static int decode_value_set(bej_jsonw* jw, bej_br* br, const bej_dict* D, bej_cluster this_cluster){
60 uint64_t count; if(!bej_read_nnint(br,&count)) return 0;
61 bej_jw_begin_obj(jw);
62 for(uint64_t i=0;i<count;i++){
63 /* Sequence (nnint). LSB=1 indicates Annotation (skipped). */
64 uint64_t S; if(!bej_read_nnint(br,&S)) return 0;
65 int is_annotation = (S & 1u) ? 1: 0;
66 uint16_t seq = (uint16_t)(S >> 1);
67
68 /* Tuple format and payload length */
69 uint8_t F; if(!bej_br_u8(br,&F)) return 0;
70 uint8_t fmt = (uint8_t)(F >> 4);
71 uint64_t L; if(!bej_read_nnint(br,&L)) return 0;
72
73 if(is_annotation){
74 /* Skip annotation payload completely */
75 if(bej_br_left(br) < (int)L) return 0;
76 br->p += (size_t)L;
77 continue;
78 }
79
80 /* Resolve property name within this cluster */
81 const bej_dict_entry* de = bej_cluster_lookup_seq(D, this_cluster, seq);
82 const char* name = (de && de->name_off)? bej_dict_name_at(D, de->name_off): NULL;
83 char tmp[32]; if(!name){ snprintf(tmp,sizeof(tmp),"seq_%u", (unsigned)seq); name = tmp; }
84
85 /* Emit key and decode value by format */
86 bej_jw_key(jw, name);
87 if(fmt==BEJ_FMT_INT){
88 if(!decode_value_int(jw, br, L)) return 0;
89 }else if(fmt==BEJ_FMT_STRING){
90 if(!decode_value_string(jw, br, L)) return 0;
91 }else if(fmt==BEJ_FMT_SET){
92 /* Descend into child cluster if known */
93 bej_cluster childc = (bej_cluster){0,0};
94 if(de && de->child_off){
95 childc.start_idx = (uint32_t)(((size_t)de->child_off - D->entries_ofs)/10);
96 childc.count = de->child_cnt;
97 }
98 if(!decode_value_set(jw, br, D, childc)) return 0;
99 }else if(fmt==BEJ_FMT_ARRAY){
100 /* Arrays: count + element tuples (we print a flat JSON array of element values) */
101 uint64_t cnt; if(!bej_read_nnint(br,&cnt)) return 0;
102 bej_jw_begin_arr(jw);
103 for(uint64_t k=0;k<cnt;k++){
104 /* Read element tuple header */
105 uint64_t Se; if(!bej_read_nnint(br,&Se)) return 0;
106 uint8_t Fe; if(!bej_br_u8(br,&Fe)) return 0;
107 uint8_t fmt_e = (uint8_t)(Fe>>4);
108 uint64_t Le; if(!bej_read_nnint(br,&Le)) return 0;
109
110 if(k>0) fputs(", ", jw->f);
111 if(fmt_e==BEJ_FMT_INT){
112 if(!decode_value_int(jw, br, Le)) return 0;
113 }else if(fmt_e==BEJ_FMT_STRING){
114 if(!decode_value_string(jw, br, Le)) return 0;
115 }else{
116 /* Unsupported element formats are skipped as null */
117 if(bej_br_left(br) < (int)Le) return 0; br->p += (size_t)Le; fputs("null", jw->f);
118 }
119 }
120 bej_jw_end_arr(jw);
121 }else if(fmt==BEJ_FMT_ENUM){
122 /* Map enum ordinal to its name via the entry's child cluster (render as JSON string) */
123 bej_br val = *br;
124 uint64_t opt_idx;
125 if(!bej_read_nnint(&val, &opt_idx)) return 0;
126 if(bej_br_left(br) < (int)L) return 0; br->p += (size_t)L;
127 const char* optname = "EnumOption";
128 if(de && de->child_off && de->child_cnt){
129 bej_cluster enumc; enumc.start_idx = (uint32_t)(((size_t)de->child_off - D->entries_ofs)/10);
130 enumc.count = de->child_cnt;
131 const bej_dict_entry* opt = bej_cluster_lookup_seq(D, enumc, (uint16_t)opt_idx);
132 if(opt && opt->name_off) optname = bej_dict_name_at(D, opt->name_off);
133 }
134 bej_jw_str(jw, optname);
135 }else{
136 /* Unsupported formats: skip payload and emit null */
137 if(bej_br_left(br) < (int)L) return 0;
138 br->p += (size_t)L;
139 fputs("null", jw->f);
140 }
141 }
142 bej_jw_end_obj(jw);
143 return 1;
144}

146/**
147 * @brief Decode a complete BEJ stream (bejEncoding + top-level tuple) and emit JSON.
148 *
149 * @param out FILE* for output JSON.
150 * @param bej Pointer to start of BEJ-encoded data.
151 * @param bej_n Length of the BEJ data.
152 * @param D Parsed schema dictionary used for names and clusters.
153 * @return 1 on success, 0 on malformed input.
154 *
155 * @note The function expects @p bej to begin with **bejEncoding** header:
156 * `version(4 LE)`, `flags(2 LE)`, `schemaClass(1)`, followed by a tuple.
157 * The top-level tuple is expected to be a **Set** whose members are emitted at JSON root.
158 */
159int bej_decode_to_json(FILE* out, const uint8_t* bej, size_t bej_n, const bej_dict* D){
160 if(!out || !bej || !D) return 0;
161
162 bej_br br; bej_br_init(&br, bej, bej_n);
163 bej_jsonw jw; bej_jw_init(&jw, out);
164
165 /* bejEncoding header */
166 if(bej_br_left(&br) < 7) return 0;
167 uint32_t ver; memcpy(&ver, br.d+br.p, 4); br.p+=4;
168 uint16_t flags; memcpy(&flags, br.d+br.p, 2); br.p+=2;
169 uint8_t schemaClass; if(!bej_br_u8(&br, &schemaClass)) return 0;
170 (void)ver; (void)flags; (void)schemaClass;
171
172 /* Root cluster (children of root entry 0) */
173 bej_cluster rootc = (bej_cluster){0,0};
174 if(D->n>0 && D->ent[0].child_off){
175 rootc.start_idx = (uint32_t)(((size_t)D->ent[0].child_off - D->entries_ofs)/10);
176 rootc.count = D->ent[0].child_cnt;
177 }
178
179 /* Parse and require a top-level Set */
180 uint64_t S; if(!bej_read_nnint(&br, &S)) return 0;
181 uint8_t F; if(!bej_br_u8(&br,&F)) return 0;
182 uint8_t fmt = (uint8_t)(F>>4);
183 uint64_t L; if(!bej_read_nnint(&br, &L)) return 0;
184 if(fmt != BEJ_FMT_SET) return 0;
185
186 /* Decode the top-level Set (decode_value_set writes the object braces) */
187 if(!decode_value_set(&jw, &br, D, rootc)) return 0;
188 fputc('\n', jw.f);
189 return 1;
190}
