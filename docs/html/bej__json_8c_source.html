26/**
27 * @brief Emit a JSON object key (with quoting/escaping) and prepare for a value.
28 * @param j JSON writer.
29 * @param k Null-terminated UTF-8 key.

31void bej_jw_key(bej_jsonw* j, const char* k){
32 if(j->need_comma) fputs(",\n", j->f); else j->need_comma=1;
33 for(int i=0;i<j->ind;i++) fputs(" ", j->f);
34 fputc('"',j->f);
35 for(const char* p=k;*p;p++){
36 if(*p=='"'||*p=='\\'){ fputc('\\',j->f); fputc(*p,j->f); }
37 else fputc(*p,j->f);
38 }
39 fputs("\": ", j->f);
40}
